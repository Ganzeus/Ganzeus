```python
import functools
import torch
import torch.nn.functional as F
import types
import time
import pprint
import contextlib
from collections import defaultdict

# =============================================================================
# 全局状态管理
# =============================================================================

# 存储每个函数的调用统计：{函数名: {"count": 次数, "total_time": 总时间}}
call_count = {}

# 跟踪装饰计数，防止同一函数被重复装饰
decorated_count = defaultdict(int)

# 备份原始函数，用于监控结束后恢复
original_functions = {}

# 跳过这些属性，避免触发PyTorch内部警告
SKIP_ATTRIBUTES = [
    'storage', 'untyped_storage', 'typed_storage',  # 存储相关属性
    'reduce_op',  # 已弃用的分布式操作
]

# =============================================================================
# 核心装饰器：函数调用监控
# =============================================================================

def count_calls(func, module_name=None):
    """
    装饰器：监控函数调用次数和执行时间
    
    Args:
        func: 被装饰的函数
        module_name: 模块名，用于生成完整的函数标识
    """
    @functools.wraps(func)
    def wrapper_count_calls(*args, **kwargs):
        # 生成唯一的函数标识符
        full_name = module_name + '.' + func.__name__
        
        # 记录开始时间（高精度计时）
        start_time = time.time()
        
        # 执行原始函数
        result = func(*args, **kwargs)
        
        # 记录结束时间
        end_time = time.time()
        elapsed_time = end_time - start_time
        
        # 更新统计信息
        if full_name not in call_count:
            call_count[full_name] = {"count": 0, "total_time": 0.0}
        
        call_count[full_name]["count"] += 1
        call_count[full_name]["total_time"] += elapsed_time
        
        return result
    
    # 标记这个函数已被装饰，避免重复装饰
    wrapper_count_calls._is_decorated = True
    wrapper_count_calls._original_func = func
    return wrapper_count_calls

# =============================================================================
# 装饰和恢复函数
# =============================================================================

def decorate_function(module, attr_name):
    """
    装饰模块中的单个函数
    
    Args:
        module: 目标模块
        attr_name: 函数名
    
    Returns:
        bool: 是否成功装饰
    """
    # 跳过可能引起问题的属性
    if attr_name in SKIP_ATTRIBUTES:
        return False
    
    if hasattr(module, attr_name):
        attr = getattr(module, attr_name)
        key = f"{module.__name__}.{attr_name}"
        
        if callable(attr):
            # 如果已经被装饰，只增加引用计数
            if hasattr(attr, '_is_decorated'):
                decorated_count[key] += 1
                return True
            
            # 首次装饰：备份原函数，创建装饰版本
            if key not in original_functions:
                original_functions[key] = attr
            
            decorated_attr = count_calls(original_functions[key], module.__name__)
            setattr(module, attr_name, decorated_attr)
            decorated_count[key] = 1
            return True
    
    return False

def restore_function(module, attr_name):
    """
    恢复模块中的单个原始函数
    
    Args:
        module: 目标模块
        attr_name: 函数名
    
    Returns:
        bool: 是否成功恢复
    """
    if attr_name in SKIP_ATTRIBUTES:
        return False
    
    key = f"{module.__name__}.{attr_name}"
    
    # 使用引用计数管理装饰状态
    if key in decorated_count:
        decorated_count[key] -= 1
        
        # 只有当没有其他地方使用时才真正恢复
        if decorated_count[key] <= 0:
            if key in original_functions:
                setattr(module, attr_name, original_functions[key])
                del decorated_count[key]
            return True
    
    return False

# =============================================================================
# 模块和类的递归装饰
# =============================================================================

def should_skip_module(module):
    """检查是否应该跳过特定模块（避免警告）"""
    module_name = getattr(module, '__name__', '')
    return module_name.startswith('torch.distributed')

def decorate_module(module, recursive=True, visited=None):
    """
    递归装饰模块中的所有函数
    
    Args:
        module: 目标模块
        recursive: 是否递归处理子模块
        visited: 已访问的模块集合（避免循环引用）
    """
    if visited is None:
        visited = set()
    
    # 避免循环引用
    module_id = id(module)
    if module_id in visited:
        return
    visited.add(module_id)
    
    # 跳过可能有问题的模块
    if should_skip_module(module):
        return
    
    # 遍历模块的所有属性
    for attr_name in dir(module):
        if attr_name in SKIP_ATTRIBUTES:
            continue
            
        try:
            attr = getattr(module, attr_name)
            
            # 装饰普通函数
            if isinstance(attr, types.FunctionType):
                decorate_function(module, attr_name)
            
            # 装饰其他可调用对象
            elif callable(attr) and not isinstance(attr, type):
                decorate_function(module, attr_name)
            
            # 递归处理子模块
            elif recursive and isinstance(attr, types.ModuleType):
                module_name = getattr(attr, '__name__', '')
                if module_name.startswith('torch') and not should_skip_module(attr):
                    decorate_module(attr, recursive=True, visited=visited)
            
            # 装饰类中的方法
            elif isinstance(attr, type):
                decorate_class(attr)
                
        except AttributeError:
            continue

def decorate_class(cls):
    """装饰类中的所有方法，包括特殊方法"""
    for attr_name in dir(cls):
        if attr_name in SKIP_ATTRIBUTES:
            continue
            
        try:
            attr = getattr(cls, attr_name)
            if isinstance(attr, types.FunctionType):
                decorate_function(cls, attr_name)
            # 特别处理运算符重载方法（如 +, -, *, / 等）
            elif attr_name in ['__add__', '__mul__', '__sub__', '__truediv__', '__matmul__', '__pow__', '__mod__']:
                decorate_function(cls, attr_name)
        except (AttributeError, TypeError):
            continue

# =============================================================================
# 恢复函数（与装饰函数对应）
# =============================================================================

def restore_module(module, recursive=True, visited=None):
    """递归恢复模块中的所有原始函数"""
    if visited is None:
        visited = set()
    
    module_id = id(module)
    if module_id in visited:
        return
    visited.add(module_id)
    
    if should_skip_module(module):
        return
        
    for attr_name in dir(module):
        if attr_name in SKIP_ATTRIBUTES:
            continue
            
        try:
            restore_function(module, attr_name)
            
            attr = getattr(module, attr_name)
            if recursive and isinstance(attr, types.ModuleType):
                module_name = getattr(attr, '__name__', '')
                if module_name.startswith('torch') and not should_skip_module(attr):
                    restore_module(attr, recursive=True, visited=visited)
            elif isinstance(attr, type):
                restore_class(attr)
                
        except AttributeError:
            continue

def restore_class(cls):
    """恢复类中的所有原始方法"""
    for attr_name in dir(cls):
        if attr_name in SKIP_ATTRIBUTES:
            continue
            
        try:
            restore_function(cls, attr_name)
        except (AttributeError, TypeError):
            continue

# =============================================================================
# 统计结果管理
# =============================================================================

def reset_counters():
    """清空所有统计数据"""
    global call_count
    call_count.clear()

def get_statistics():
    """获取当前的统计结果"""
    return call_count

def print_statistics(sort_by="count", top_n=None):
    """
    格式化打印统计结果
    
    Args:
        sort_by: 排序字段 ("count" 或 "total_time")
        top_n: 只显示前N个结果
    """
    if not call_count:
        print("No statistics available.")
        return
        
    # 按指定字段排序
    sorted_stats = sorted(call_count.items(), 
                          key=lambda x: x[1][sort_by], 
                          reverse=True)
    
    if top_n is not None:
        sorted_stats = sorted_stats[:top_n]
    
    # 格式化输出
    print(f"{'Function Name':<50} {'Count':<10} {'Total Time (s)':<15} {'Avg Time (s)':<15}")
    print("-" * 90)
    for name, stats in sorted_stats:
        avg_time = stats["total_time"] / stats["count"] if stats["count"] > 0 else 0
        print(f"{name:<50} {stats['count']:<10} {stats['total_time']:<15.6f} {avg_time:<15.6f}")

# 上下文管理器：临时监控
# =============================================================================

class Monitor:
    """上下文管理器类，用于临时启用函数监控"""
    
    def __init__(self, modules=None, reset=False):
        """
        Args:
            modules: 要监控的模块列表，默认监控整个torch模块
            reset: 进入监控前是否重置统计数据
        """
        self.modules = modules if modules is not None else [torch]
        self.reset = reset
    
    def __enter__(self):
        """进入上下文：开始监控"""
        if self.reset:
            reset_counters()
        
        # 装饰指定模块
        for module in self.modules:
            decorate_module(module)
        
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出上下文：停止监控，恢复原始函数"""
        for module in self.modules:
            restore_module(module)

@contextlib.contextmanager
def monitor(modules=None, reset=False):
    """
    便捷的上下文管理器函数
    
    使用方式:
        with monitor():
            # 这里的PyTorch操作会被监控
            y = model(x)
    """
    with Monitor(modules=modules, reset=reset):
        yield
```

## 装饰器基础

### 一、装饰器基础概念

装饰器是**Python的内置特性**，不是第三方库。它本质上就是一个**接受函数作为参数并返回新函数的函数**。

装饰器本质上就是**函数式编程中的高阶函数**概念在Python中的实现，通过这种方式可以在不修改原代码的情况下增强功能。

#### 最简单的装饰器例子

```python
# 1. 最基础的装饰器
def my_decorator(func):
    def wrapper():
        print("函数执行前")
        result = func()  # 执行原函数
        print("函数执行后")
        return result
    return wrapper

# 使用方式1：手动装饰
def say_hello():
    print("Hello!")

say_hello = my_decorator(say_hello)  # 手动装饰
say_hello()

# 使用方式2：@语法糖（推荐）
@my_decorator
def say_goodbye():
    print("Goodbye!")

say_goodbye()
```

输出：

```
函数执行前
Hello!
函数执行后
函数执行前
Goodbye!
函数执行后
```



### 二、装饰器的演进过程

#### 第一步：基础计时装饰器

```python
import time

def timer(func):
    def wrapper():
        start = time.time()
        result = func()
        end = time.time()
        print(f"{func.__name__} 耗时: {end - start:.6f}秒")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "完成"

slow_function()  # 输出: slow_function 耗时: 1.001234秒
```

#### 第四步：收集统计信息

```python
import functools
import time

# 全局统计字典
stats = {}

def monitor(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        
        # 收集统计信息
        func_name = func.__name__
        if func_name not in stats:
            stats[func_name] = {"count": 0, "total_time": 0}
        
        stats[func_name]["count"] += 1
        stats[func_name]["total_time"] += (end - start)
        
        return result
    return wrapper

@monitor
def test_func():
    time.sleep(0.1)

# 调用几次
test_func()
test_func()
print(stats)  # {'test_func': {'count': 2, 'total_time': 0.2}}
```

#### 思路3：上下文管理器

```python
class Monitor:
    def __init__(self):
        self.original_functions = {}
    
    def __enter__(self):
        # 开始监控：替换函数
        self.original_functions['torch.add'] = torch.add
        torch.add = self.monitored_add
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        # 结束监控：恢复原函数
        torch.add = self.original_functions['torch.add']
    
    def monitored_add(self, *args, **kwargs):
        print("监控中...")
        return self.original_functions['torch.add'](*args, **kwargs)

# 使用
with Monitor():
    result = torch.add(x, y)  # 被监控
    
result = torch.add(x, y)  # 不被监控
```


### 完整的装饰器学习路径

#### 无参装饰器

```python
def log_calls(func):
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

@log_calls
def greet(name):
    return f"Hello {name}"
```



#### 带参数的装饰器

```python
def repeat(times):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(3)  # 相当于 repeat(3)(say_hello)
def say_hello():
    print("Hello!")

say_hello()  # 打印3次Hello!
```



#### 类装饰器

```python
class CallCounter:
    def __init__(self, func):
        self.func = func
        self.count = 0
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"第{self.count}次调用")
        return self.func(*args, **kwargs)

@CallCounter
def test():
    print("测试")

test()  # 第1次调用 测试
test()  # 第2次调用 测试
```
